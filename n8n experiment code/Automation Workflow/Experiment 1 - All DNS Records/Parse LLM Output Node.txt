// Robust parser for Ollama /api/chat output.
// Ollama sometimes returns extra wrappers like leading "(" or trailing ")".
// This extracts the first {...} JSON object and parses it safely.

const content =
  $json.message?.content ??
  $json.response?.message?.content ??
  $json.body?.message?.content ??
  $json.data?.message?.content ??
  $json.choices?.[0]?.message?.content ??
  "";

function extractFirstJsonObject(text) {
  if (!text) return null;

  // Trim common leading/trailing junk
  let s = String(text).trim();

  // Find first '{' and then scan until matching '}' depth returns to 0
  const start = s.indexOf("{");
  if (start === -1) return null;

  let depth = 0;
  let inString = false;
  let escape = false;

  for (let i = start; i < s.length; i++) {
    const ch = s[i];

    if (inString) {
      if (escape) {
        escape = false;
      } else if (ch === "\\") {
        escape = true;
      } else if (ch === "\"") {
        inString = false;
      }
      continue;
    } else {
      if (ch === "\"") {
        inString = true;
        continue;
      }
      if (ch === "{") depth++;
      if (ch === "}") depth--;

      if (depth === 0) {
        return s.slice(start, i + 1);
      }
    }
  }
  return null;
}

let parsed;
try {
  const jsonStr = extractFirstJsonObject(content);
  if (!jsonStr) throw new Error("no_json_object_found");
  parsed = JSON.parse(jsonStr);
} catch (e) {
  // If parse fails, still produce a safe document id so Post-to-Elastic won't break
  parsed = {
    run_id: `parse_error_run_${$itemIndex}`,
    doc_id: `parse_error_doc_${$itemIndex}`,
    llm_label: "benign",
    confidence: 0.01,
    reason: "parse_error: could not parse JSON output"
  };
}

// Prefer echoed IDs from the model output; ensure non-null
const run_id = String(parsed.run_id ?? `missing_run_${$itemIndex}`);
const doc_id = String(parsed.doc_id ?? `missing_doc_${$itemIndex}`);

// Enforce benign|malicious only
let llm_label = String(parsed.llm_label ?? "benign").toLowerCase().trim();
if (llm_label !== "benign" && llm_label !== "malicious") llm_label = "benign";

// Clamp confidence to [0.0, 1.0]
let confidence = Number(parsed.confidence);
if (!Number.isFinite(confidence)) confidence = 0.01;
confidence = Math.max(0, Math.min(1, confidence));

// Normalize reason
let reason = String(parsed.reason ?? "").replace(/\s+/g, " ").trim().slice(0, 500);
if (!reason) reason = "no_reason_provided";

return [
  {
    json: {
      run_id,
      doc_id,
      llm_label,
      confidence,
      reason
    }
  }
];
